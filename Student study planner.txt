let focusTotal=25*60,focusLeft=25*60,focusRunning=false,focusTimer=null;
function enterFocus(){
  const mins=parseInt(document.getElementById('set-focus')?.value||25)||25;
  focusTotal=mins*60; focusLeft=focusTotal; focusRunning=false;
  clearInterval(focusTimer);
  document.getElementById('focus-play-btn').textContent='▶ Start';
  document.getElementById('focus-bar').style.width='100%';
  updateFocusClock();
  document.getElementById('focus-overlay').classList.add('open');
}
function exitFocus(){clearInterval(focusTimer);focusRunning=false;document.getElementById('focus-overlay').classList.remove('open');showToast('🎯 Focus session ended!');}
function toggleFocusTimer(){
  if(focusRunning){clearInterval(focusTimer);focusRunning=false;document.getElementById('focus-play-btn').textContent='▶ Resume';}
  else{
    focusRunning=true;document.getElementById('focus-play-btn').textContent='⏸ Pause';
    focusTimer=setInterval(()=>{
      if(focusLeft>0){focusLeft--;updateFocusClock();document.getElementById('focus-bar').style.width=(focusLeft/focusTotal*100)+'%';}
      else{clearInterval(focusTimer);focusRunning=false;exitFocus();showToast('🎉 Session complete! Great work!');}
    },1000);
  }
}
function updateFocusClock(){const m=String(Math.floor(focusLeft/60)).padStart(2,'0'),s=String(focusLeft%60).padStart(2,'0');document.getElementById('focus-timer').textContent=`${m}:${s}`;}

function revProg(s){return s.topics.length?Math.round(s.topics.filter(t=>doneTopics.has(t)).length/s.topics.length*100):0;}
function updateRevOverall(){
  const all=[...revisionData.flatMap(s=>s.topics)],done=all.filter(t=>doneTopics.has(t)).length;
  const pct=all.length?Math.round(done/all.length*100):0;
  document.getElementById('rev-overall').textContent=`Overall: ${pct}% revised`;
  document.getElementById('st-rev').textContent=pct+'%';
}
function renderRevision(){
  document.getElementById('revision-grid').innerHTML=revisionData.map((s,si)=>{
    const p=revProg(s);
    return `<div class="subject-card">
      <div class="sc-head"><h4>${s.subj}</h4><button class="add-sm-btn" onclick="openModal('topic',${si})">+ Topic</button></div>
      <div class="progress-bar"><div class="progress-fill" style="width:${p}%"></div></div>
      <div class="progress-meta"><span>${p}% revised</span><span>${s.topics.filter(t=>doneTopics.has(t)).length}/${s.topics.length} topics</span></div>
      <div class="topic-list">
        ${s.topics.map((t,ti)=>`<div class="topic-item">

          </div>
          <button class="topic-del" onclick="delTopic(${si},${ti})">✕</button>

      </div>
    </div>`;
  }).join('');

}
